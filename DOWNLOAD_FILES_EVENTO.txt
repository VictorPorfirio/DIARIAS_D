create or replace PROCEDURE DOWNLOAD_FILES_EVENTO(
		p_file in number -- id do evento se for evento, se for outros, id do arquivo

		) AS
        v_mime  VARCHAR2(48);
        v_length  NUMBER;
        v_file_name VARCHAR2(2000);
        Lob_loc  BLOB;
BEGIN
-- Arquivos de eventos = possuem pr√≥pria tabela de arquivos

		SELECT MIMETYPE_TEMP, ARQUIVO, FILENAME_TEMP,DBMS_LOB.GETLENGTH(ARQUIVO)
					INTO v_mime,lob_loc,v_file_name,v_length
					FROM DIARIAS_ARQUIVO_EVENTO
					WHERE codigo = p_file; 
				--
				-- set up HTTP header
				--
				-- use an NVL around the mime type and 
				-- if it is a null set it to application/octect
				-- application/octect may launch a download window from windows
				owa_util.mime_header( nvl(v_mime,'application/octet'), FALSE );
	 
					-- set the size so the browser knows how much to download
					htp.p('Content-length: ' || v_length);
					-- the filename will be used by the browser if the users does a save as
					htp.p('Content-Disposition:  attachment; filename="'||replace(replace(substr(v_file_name,instr(v_file_name,'/')+1),chr(10),null),chr(13),null)|| '"');
					-- close the headers            
					owa_util.http_header_close;
					-- download the BLOB
					wpg_docload.download_file( Lob_loc ); 
	

	
end;
